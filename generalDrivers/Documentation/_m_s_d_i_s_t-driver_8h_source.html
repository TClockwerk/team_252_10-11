00068 /**
00069  * Read the distance from the sensor
00070  * @param link the sensor port number
00071  * @return distance to object or -1 if an error occurred
00072  */
00073 int MSDISTreadDist(tSensors link) {
00074   memset(MSDIST_I2CRequest, 0, sizeof(tByteArray));
00075 
00076   MSDIST_I2CRequest.arr[0] = 2;               // Number of bytes in I2C command
00077   MSDIST_I2CRequest.arr[1] = MSDIST_I2C_ADDR; // I2C address of accel sensor
00078   MSDIST_I2CRequest.arr[2] = MSDIST_DIST;     // Set write address to sensor mode register
00079 
00080   if (!writeI2C(link, MSDIST_I2CRequest, 2))
00081     return -1;
00082 
00083   if (!readI2C(link, MSDIST_I2CReply, 2))
00084     return -1;
00085 
00086   return (0x00FF & MSDIST_I2CReply.arr[0]) + ((0x00FF & MSDIST_I2CReply.arr[1]) <<8);
00087 }
