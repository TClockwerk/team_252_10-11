00064 /**
00065  * Send a direct command to the PFMate sensor
00066  *
00067  * Note: this is an internal command and should not be called directly.
00068  * @param link the PFMate port number
00069  * @param command the command to be sent
00070  * @return true if no error occured, false if it did
00071  */
00072 bool _MSPFMsendCommand(tSensors link, byte command) {
00073   memset(MSPFM_I2CRequest, 0, sizeof(tByteArray));
00074   MSPFM_I2CRequest.arr[0] = 3;
00075   MSPFM_I2CRequest.arr[1] = MSPFM_I2C_ADDR;
00076   MSPFM_I2CRequest.arr[2] = MSPFM_CMD;
00077   MSPFM_I2CRequest.arr[3] = command;
00078 
00079   return writeI2C(link, MSPFM_I2CRequest, 0);
00080 }
