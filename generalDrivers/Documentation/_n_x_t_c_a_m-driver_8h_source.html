00199 /**
00200  * Check if two blobs can be merged into one.  blob1 will be replaced with the
00201  * new merged blob and blob2 will be destroyed.
00202  *
00203  * Note: this is an internal function and should not be called directly.
00204  * @param blob1 the index number of the first blob
00205  * @param blob2 the index number of the second blob
00206  * @param nblobs the number of blobs
00207  * @param blobs the array of blobs
00208  * @return the number of blobs detected
00209  */
00210 int _mergeBlobs(int blob1, int blob2, int nblobs, blob_array &blobs) {
00211   int _blob1_center;
00212   int _blob2_center;
00213   int _blob1_proj;
00214   int _blob2_proj;
00215   bool _overlapx = false;
00216   bool _overlapy = false;
00217 
00218   // If either pf the blobs are size 0, just skip them
00219   if (blobs[blob1].size == 0 || blobs[blob2].size == 0)
00220     return nblobs;
00221 
00222   // If the colours don't match, don't _merge them
00223   if (blobs[blob1].colour != blobs[blob2].colour)
00224     return nblobs;
00225 
00226   // Find the center of the top sides of each blob and their projections onto
00227   // the X plane from center to right corner
00228   _blob1_center = SIDE_CENTER(blobs[blob1].x1, blobs[blob1].x2);
00229   _blob2_center = SIDE_CENTER(blobs[blob2].x1, blobs[blob2].x2);
00230   _blob1_proj = blobs[blob1].x2 - _blob1_center;
00231   _blob2_proj = blobs[blob2].x2 - _blob2_center;
00232 
00233   // Check if the projections overlap
00234   if ((_blob1_proj + _blob2_proj) > abs(_blob1_center - _blob2_center))
00235     _overlapx = true;
00236 
00237   // Find the center of the left sides of each blob and their projections onto
00238   // the Y plane from center to bottom corner
00239   _blob1_center = SIDE_CENTER(blobs[blob1].y1, blobs[blob1].y2);
00240   _blob2_center = SIDE_CENTER(blobs[blob2].y1, blobs[blob2].y2);
00241   _blob1_proj = blobs[blob1].y2 - _blob1_center;
00242   _blob2_proj = blobs[blob2].y2 - _blob2_center;
00243 
00244   // Check if the projections overlap
00245   if ((_blob1_proj + _blob2_proj) > abs(_blob1_center - _blob2_center))
00246     _overlapy = true;
00247 
00248   // If both projections are overlapping, then the blobs are _merged.
00249   // Find the smallest bounding box for both blobs and replace blob1
00250   // with this new information.  Blob2 gets set to 0 size and
00251   // recursively replaced by the next in line.
00252   if ((_overlapx == true) && (_overlapy == true)) {
00253     blobs[blob1].x1 = min(blobs[blob1].x1, blobs[blob2].x1);
00254     blobs[blob1].y1 = min(blobs[blob1].y1, blobs[blob2].y1);
00255     blobs[blob1].x2 = max(blobs[blob1].x2, blobs[blob2].x2);
00256     blobs[blob1].y2 = max(blobs[blob1].y2, blobs[blob2].y2);
00257     blobs[blob1].size = abs(blobs[blob1].x2 - blobs[blob1].x1) * abs(blobs[blob1].y2 - blobs[blob1].y1);
00258 
00259     for (int _i = blob2; _i < nblobs - 1; _i++) {
00260       memcpy(blobs[_i], blobs[_i+1], sizeof(blob));
00261     }
00262 
00263     nblobs--;
00264     memset(blobs[nblobs], 0, sizeof(blob));
00265     still_merging = true;
00266   }
00267   _sortBlobs(nblobs, blobs);
00268 
00269   // return the new number of blobs
00270   return nblobs;
00271 }

00282   int i, j;
00283 
00284   // Outer boundary definition
