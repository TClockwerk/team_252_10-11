00111 /**
00112  * Tell the servo to move to the specified position.  It will use whatever
00113  * speed is already set in the speed register.  Use NXTServoSetSpeed() to change
00114  * this setting or use NXTServoSetPos() to set both at once.
00115  *
00116  * The position argument is clipped at 500 2500 to prevent damage to the servo.
00117  * @param link the NXTServo port number
00118  * @param servochan the servo channel to use
00119  * @param position the position to move the servo to
00120  * @return true if no error occured, false if it did
00121  */
00122 bool NXTServoSetPos(tSensors link, ubyte servochan, int position) {
00123   if (position < 500) position = 500;
00124   if (position > 2500) position = 2500;
00125 
00126 
00127   // set the position register and tell NXTServo to move the servo
00128   memset(NXTSERVO_I2CRequest, 0, sizeof(tByteArray));
00129   NXTSERVO_I2CRequest.arr[0] = 4;
00130   NXTSERVO_I2CRequest.arr[1] = NXTSERVO_I2C_ADDR;
00131   NXTSERVO_I2CRequest.arr[2] = NXTSERVO_POS_CHAN1 + ((servochan - 1) * 2);
00132   NXTSERVO_I2CRequest.arr[3] = position & 0x00FF;
00133   NXTSERVO_I2CRequest.arr[4] = (position >> 8) & 0x00FF;
00134 
00135   return writeI2C(link, NXTSERVO_I2CRequest, 0);
00136 }

00205   int position = 0;
00206 
00207   memset(NXTSERVO_I2CRequest, 0, sizeof(tByteArray));
00208 
00209   NXTSERVO_I2CRequest.arr[0] = 2;                 // Message size
00210   NXTSERVO_I2CRequest.arr[1] = NXTSERVO_I2C_ADDR; // I2C Address
00211   NXTSERVO_I2CRequest.arr[2] = NXTSERVO_POS_CHAN1 + ((servochan - 1) * 2);
00212 
00213   if (!writeI2C(link, NXTSERVO_I2CRequest, 2))
00214     return false;
00215 
00216   if (!readI2C(link, NXTSERVO_I2CReply, 2))
00217     return false;
00218 
00219   position = NXTSERVO_I2CReply.arr[1] * 256 + NXTSERVO_I2CReply.arr[0];
00220 
00221   return position;
